iptables -t nat -A POSTROUTING -s 10.1.1.2 -j MASQUERADE
iptables -A FORWARD -i dtap0 -s 10.1.1.2 -j ACCEPT
iptables -A FORWARD -o dtap0 -d 10.1.1.2 -j ACCEPT 

# bridge
ip link add name br0 type bridge

ip netns add ns1
ip link set dev vethp0 netns ns1
